CREATE DATABASE yektanet
CREATE TABLE Ad(
	ID		INT						NOT NULL,
	title	VARCHAR (500)			NOT NULL,
	imgUrl	VARCHAR (1000)			NOT NULL,
	link	VARCHAR (1000)			NOT NULL,
	clicks	INT,
	views INT,
	PRIMARY KEY (ID)
);

CREATE TABLE Advertiser(
	ID		INT						NOT NULL,
	name	VARCHAR (100)			NOT NULL,
	clicks	INT,
	views INT,
	PRIMARY KEY (ID)
);
